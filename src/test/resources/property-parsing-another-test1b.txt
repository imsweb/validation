// using list.get() instead of the array index...
for (i = 0; i < lines.size() - 1; i++)
    for (j = i+1; j < lines.size(); j++) {
        def dx1 = Functions.asInt(lines.get(i).dateOfDiagnosisYear)
        def dx2 = Functions.asInt(lines.get(j).dateOfDiagnosisYear)
        def hist1 = lines[i].histologicTypeIcdO3 == null ? '' : lines.get(i).histologicTypeIcdO3
        def hist2 = lines[j].histologicTypeIcdO3 == null ? '' : lines.get(j).histologicTypeIcdO3
        def beh1 = lines[i].behaviorCodeIcdO3 == null ? '' : lines.get(i).behaviorCodeIcdO3
        def beh2 = lines[j].behaviorCodeIcdO3 == null ? '' : lines.get(j).behaviorCodeIcdO3
        def icd1 = hist1 + '/' + beh1
        def icd2 = hist2 + '/' + beh2
        if ((dx1 >= 2010 || dx2 >= 2010) && hemato_same_primaries.containsKey(icd1) && hemato_same_primaries[icd1].contains(icd2)) {
            Functions.forceFailureOnEntity(binding, lines.get(i))
            Functions.forceFailureOnEntity(binding, lines.get(j))
            return false
        }
    }
return true
